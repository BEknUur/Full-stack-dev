# 1. Вывод списка всех переменных окружения
printenv

# 2. Проверка значения переменной PATH
echo $PATH

# 3. Проверка значения переменной HOME
echo $HOME

# 4. Проверка значения переменной USER
echo $USER

# 5. Проверка значения переменной SHELL
echo $SHELL

# 6. Установка новой переменной окружения MY_VAR
export MY_VAR="Значение моей переменной"

# 7. Проверка значения новой переменной окружения MY_VAR
echo $MY_VAR
